    <script>

        let conversations = JSON.parse(localStorage.getItem('claudeConversations') || '[]');

        let currentTags = [];

        function updateStats() {

            const today = new Date().toDateString();

            const weekAgo = new Date(Date.now() - 7 * 24 * 60 * 60 * 1000);

           

            const todayCount = conversations.filter(c => new Date(c.date).toDateString() === today).length;

            const weekCount = conversations.filter(c => new Date(c.date) >= weekAgo).length;

           

            document.getElementById('totalCount').textContent = conversations.length;

            document.getElementById('todayCount').textContent = todayCount;

            document.getElementById('weekCount').textContent = weekCount;

        }

        function renderConversations(conversationsToShow = conversations) {

            const container = document.getElementById('conversationsList');

           

            if (conversationsToShow.length === 0) {

                container.innerHTML = '<div style="text-align: center; color: #6b7280; padding: 40px;">No conversations found.</div>';

                return;

            }

           

            container.innerHTML = conversationsToShow.map(conv => `

                <div class="conversation-item" onclick="viewConversation('${conv.id}')">

                    <div class="conversation-title">${conv.title}</div>

                    <div class="conversation-date">${new Date(conv.date).toLocaleDateString()} • ${conv.category}</div>

                    <div class="conversation-summary">${conv.summary}</div>

                    <div class="conversation-tags">

                        ${conv.tags.map(tag => `<span class="conversation-tag">${tag}</span>`).join('')}

                    </div>

                </div>

            `).join('');

        }

        function addTag() {

            const tagInput = document.getElementById('tags');

            const tag = tagInput.value.trim().toLowerCase();

           

            if (tag && !currentTags.includes(tag)) {

                currentTags.push(tag);

                renderTags();

                tagInput.value = '';

            }

        }

        function removeTag(tag) {

            currentTags = currentTags.filter(t => t !== tag);

            renderTags();

        }

        function renderTags() {

            const container = document.getElementById('tagContainer');

            container.innerHTML = currentTags.map(tag => `

                <span class="tag">

                    ${tag}

                    <span class="tag-remove" onclick="removeTag('${tag}')">×</span>

                </span>

            `).join('');

        }

        function saveConversation() {

            const title = document.getElementById('title').value.trim();

            const summary = document.getElementById('summary').value.trim();

            const content = document.getElementById('content').value.trim();

            const category = document.getElementById('category').value;

           

            if (!title || !summary || !content) {

                alert('Please fill in all required fields');

                return;

            }

           

            const conversation = {

                id: Date.now().toString(),

                title,

                summary,

                content,

                category,

                tags: [...currentTags],

                date: new Date().toISOString()

            };

           

            conversations.unshift(conversation);

            localStorage.setItem('claudeConversations', JSON.stringify(conversations));

           

            // Clear form

            document.getElementById('title').value = '';

            document.getElementById('summary').value = '';

            document.getElementById('content').value = '';

            document.getElementById('category').value = 'coding';

            currentTags = [];

            renderTags();

           

            updateStats();

            renderConversations();

           

            alert('Conversation saved successfully!');

        }

        function viewConversation(id) {

            const conv = conversations.find(c => c.id === id);

            if (conv) {

                const popup = window.open('', '_blank', 'width=800,height=600');

                popup.document.write(`

                    <html>

                        <head>

                            <title>${conv.title}</title>

                            <style>

                                body { font-family: Arial, sans-serif; padding: 20px; line-height: 1.6; }

                                .header { border-bottom: 2px solid #eee; padding-bottom: 20px; margin-bottom: 20px; }

                                .content { white-space: pre-wrap; }

                                .tag { background: #e0e7ff; color: #4f46e5; padding: 2px 8px; border-radius: 12px; font-size: 12px; margin-right: 5px; }

                            </style>

                        </head>

                        <body>

                            <div class="header">

                                <h1>${conv.title}</h1>

                                <p><strong>Date:</strong> ${new Date(conv.date).toLocaleDateString()}</p>

                                <p><strong>Category:</strong> ${conv.category}</p>

                                <p><strong>Summary:</strong> ${conv.summary}</p>

                                <p><strong>Tags:</strong> ${conv.tags.map(tag => `<span class="tag">${tag}</span>`).join('')}</p>

                            </div>

                            <div class="content">${conv.content}</div>

                        </body>

                    </html>

                `);

            }

        }

        function filterConversations() {

            const query = document.getElementById('searchBox').value.toLowerCase();

            const filtered = conversations.filter(conv =>

                conv.title.toLowerCase().includes(query) ||

                conv.summary.toLowerCase().includes(query) ||

                conv.tags.some(tag => tag.includes(query)) ||

                conv.category.toLowerCase().includes(query)

            );

            renderConversations(filtered);

        }

        function exportToMarkdown() {

            let markdown = '# Claude Conversations\n\n';

           

            conversations.forEach(conv => {

                markdown += `## ${conv.title}\n\n`;

                markdown += `**Date:** ${new Date(conv.date).toLocaleDateString()}\n`;

                markdown += `**Category:** ${conv.category}\n`;

                markdown += `**Tags:** ${conv.tags.join(', ')}\n\n`;

                markdown += `**Summary:** ${conv.summary}\n\n`;

                markdown += `### Full Conversation\n\n`;

                markdown += `${conv.content}\n\n`;

                markdown += '---\n\n';

            });

           

            downloadFile('claude-conversations.md', markdown);

        }

        function exportToJSON() {

            downloadFile('claude-conversations.json', JSON.stringify(conversations, null, 2));

        }

        function downloadFile(filename, content) {

            const blob = new Blob([content], { type: 'text/plain' });

            const url = URL.createObjectURL(blob);

            const a = document.createElement('a');

            a.href = url;

            a.download = filename;

            document.body.appendChild(a);

            a.click();

            document.body.removeChild(a);

            URL.revokeObjectURL(url);

        }

        });
